Projet de Référendum Sécurisé

Dans le cadre de notre SAE "Systèmes distribués", j'ai dû implémenter l'algorithme de chiffrement El Gamal pour sécuriser un système de vote électronique. Ce choix algorithmique était crucial car il devait garantir à la fois la confidentialité des votes et la possibilité de vérifier l'intégrité du processus électoral sans compromettre l'anonymat des votants.

Contraintes techniques

  • • Préservation de l'anonymat absolu des votes
  • • Impossibilité de modifier les votes sans détection
  • • Vérification des résultats sans déchiffrer les votes individuels
  • • Performance adéquate pour traiter un grand nombre de votes

Solutions apportées

  • • Utilisation d'El Gamal pour ses propriétés homomorphiques
  • • Implémentation d'une architecture à clé publique/privée distribuée
  • • Optimisation des calculs modulaires pour améliorer la performance
  • • Séparation entre le serveur de vote et le système de dépouillement

Voici un extrait de mon implémentation de l'algorithme El Gamal pour le chiffrement des votes : 

public class ElGamalEncryption {
    private BigInteger p; // Grand nombre premier
    private BigInteger g; // Générateur
    private BigInteger y; // Clé publique
    private BigInteger x; // Clé privée (seulement pour le déchiffreur)

    // Génération des paramètres
    public ElGamalEncryption(int bitLength) {
        SecureRandom random = new SecureRandom();
        p = BigInteger.probablePrime(bitLength, random);
        g = findGenerator(p);
        x = new BigInteger(bitLength - 1, random);
        y = g.modPow(x, p);
    }

    // Chiffrement d'un vote (message = 1 pour Oui, 0 pour Non)
    public ElGamalCiphertext encrypt(int message) {
        SecureRandom random = new SecureRandom();
        BigInteger k = new BigInteger(p.bitLength() - 1, random);
        BigInteger alpha = g.modPow(k, p);

        // Conversion du vote en BigInteger
        BigInteger m = BigInteger.valueOf(message);

        // Beta = y^k * m mod p
        BigInteger beta = (y.modPow(k, p).multiply(g.modPow(m, p))).mod(p);

        return new ElGamalCiphertext(alpha, beta);
    }

    // Classe interne pour représenter un texte chiffré
    public static class ElGamalCiphertext {
        public final BigInteger alpha;
        public final BigInteger beta;

        public ElGamalCiphertext(BigInteger alpha, BigInteger beta) {
            this.alpha = alpha;
            this.beta = beta;
        }
    }

    // Déchiffrement (utilisé uniquement lors du dépouillement)
    public int decrypt(ElGamalCiphertext ciphertext) {
        // Calcul de alpha^x
        BigInteger alphaX = ciphertext.alpha.modPow(x, p);
        // Calcul de son inverse modulaire
        BigInteger alphaXInv = alphaX.modInverse(p);
        // Déchiffrement: m = beta * (alpha^x)^(-1) mod p
        BigInteger m = ciphertext.beta.multiply(alphaXInv).mod(p);

        // Pour un vote binaire, on vérifie si m correspond à g^0 ou g^1
        if (m.equals(BigInteger.ONE)) {
            return 0; // Non
        } else if (m.equals(g)) {
            return 1; // Oui
        } else {
            throw new IllegalArgumentException("Vote invalide détecté");
        }
    }
}

La structure a été optimisée pour garantir une séparation stricte entre les identités des électeurs et leurs votes, tout en permettant la vérification qu'un utilisateur n'a voté qu'une seule fois.

-- Schéma de base de données pour le système de vote
CREATE TABLE Referendum (
    id INT PRIMARY KEY AUTO_INCREMENT,
    titre VARCHAR(255) NOT NULL,
    description TEXT NOT NULL,
    date_debut DATETIME NOT NULL,
    date_fin DATETIME NOT NULL,
    question VARCHAR(500) NOT NULL,
    est_actif BOOLEAN DEFAULT FALSE,
    public_key TEXT NOT NULL -- Clé publique pour le chiffrement
);

CREATE TABLE Utilisateur (
    id INT PRIMARY KEY AUTO_INCREMENT,
    login VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL, -- Hash du mot de passe (jamais en clair)
    nom VARCHAR(100) NOT NULL,
    prenom VARCHAR(100) NOT NULL,
    role ENUM('VOTANT', 'ADMIN') NOT NULL DEFAULT 'VOTANT'
);

-- Table de participation qui stocke simplement qui a voté (sans lien avec le vote)
CREATE TABLE Participation (
    utilisateur_id INT NOT NULL,
    referendum_id INT NOT NULL,
    date_vote TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (utilisateur_id, referendum_id),
    FOREIGN KEY (utilisateur_id) REFERENCES Utilisateur(id),
    FOREIGN KEY (referendum_id) REFERENCES Referendum(id)
);

-- Table de votes qui ne contient aucun lien avec l'identité du votant
CREATE TABLE Vote (
    id INT PRIMARY KEY AUTO_INCREMENT,
    referendum_id INT NOT NULL,
    vote_chiffre_alpha TEXT NOT NULL, -- Première partie du vote chiffré avec El Gamal
    vote_chiffre_beta TEXT NOT NULL,  -- Deuxième partie du vote chiffré avec El Gamal
    date_creation TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (referendum_id) REFERENCES Referendum(id)
);

-- Index pour optimiser les requêtes fréquentes
CREATE INDEX idx_vote_referendum ON Vote(referendum_id);
CREATE INDEX idx_referendum_dates ON Referendum(date_debut, date_fin);

-- Vérifier si un utilisateur a déjà voté (optimisé)
CREATE PROCEDURE a_deja_vote(IN p_utilisateur_id INT, IN p_referendum_id INT, OUT resultat BOOLEAN)
BEGIN
    DECLARE compte INT DEFAULT 0;

    SELECT COUNT(*) INTO compte
    FROM Participation
    WHERE utilisateur_id = p_utilisateur_id AND referendum_id = p_referendum_id;

    IF compte > 0 THEN
        SET resultat = TRUE;
    ELSE
        SET resultat = FALSE;
    END IF;
END;
